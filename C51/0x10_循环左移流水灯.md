## 源代码   
---

```c
#include <reg51.h>   //包含51单片机寄存器定义的头文件
#include <intrins.h> //包含函数isalpha（）声明的头文件
/*************************************************
函数功能：延时约150ms
*************************************************/
void delay150ms(void) // 误差 0us
{
  unsigned char a, b, c;
  for (c = 3; c > 0; c--)
    for (b = 116; b > 0; b--)
      for (a = 214; a > 0; a--)
        ;
}

/*************************************************
函数功能：主函数
*************************************************/
void main(void)
{
  P3 = 0xfe; // P3=1111 1110B
  while (1)
  {
    P3 = _crol_(P3, 1); // 将P3的二进制位循环左移1位后再赋给P3
    delay150ms();       // 调用延时函数
  }
}
```
