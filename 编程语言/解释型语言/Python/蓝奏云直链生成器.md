import tkinter as tk
from tkinter import scrolledtext
import pyperclip

def replace_link():
    input_link = input_entry.get()
    replaced_link = ""
    # 在这里添加链接替换的逻辑
    if input_link.startswith("http"):
        replaced_link = "https://www.yuanxiapi.cn/api/lanzou/?url={}&type=down".format(input_link)

        pyperclip.copy(replaced_link)
    output_text.delete(1.0, tk.END)
    output_text.insert(tk.END, replaced_link)

def copy_markdown_link():
    markdown_link = "[" + "xxx" + "](" + output_text.get(1.0, tk.END).strip() + ")"
    pyperclip.copy(markdown_link)

# 创建主窗口
root = tk.Tk()
root.title("蓝奏云直链生成工具")

# 创建输入框
input_entry = tk.Entry(root, width=34)
input_entry.grid(row=0, column=0, columnspan=2, padx=10, pady=10)

# 创建替换按钮
replace_button = tk.Button(root, text="生成直链", command=replace_link)
replace_button.grid(row=1, column=0, columnspan=2, padx=10, pady=5)

# 创建输出框
output_text = scrolledtext.ScrolledText(root, width=30, height=1)
output_text.grid(row=2, column=0, columnspan=2, padx=10, pady=10)

# 创建复制链接按钮
copy_button = tk.Button(root, text="复制URL链接", command=lambda: pyperclip.copy(output_text.get(1.0, tk.END).strip()))
copy_button.grid(row=3, column=0, padx=10, pady=10, sticky="EW")

# 创建复制Markdown链接按钮
copy_markdown_button = tk.Button(root, text="复制Markdown链接", command=copy_markdown_link)
copy_markdown_button.grid(row=3, column=1, padx=10, pady=10, sticky="EW")

# 设置所有按钮和文本框居中
for child in root.winfo_children():
    child.grid_configure(padx=5, pady=5)

# 运行主循环
root.mainloop()
